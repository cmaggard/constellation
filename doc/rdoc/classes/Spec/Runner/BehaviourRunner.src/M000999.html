      def run(paths, exit_when_done)
        prepare!(paths)
        begin
          run_behaviours
        rescue Interrupt
        ensure
          report_end
        end
        failure_count = report_dump
        
        heckle if(failure_count == 0 && !@options.heckle_runner.nil?)
        
        if(exit_when_done)
          exit_code = (failure_count == 0) ? 0 : 1
          exit(exit_code)
        end
        failure_count
      end
