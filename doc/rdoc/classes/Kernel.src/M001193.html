  def describe(*args, &block)
    raise ArgumentError if args.empty?
    args << {} unless Hash === args.last
    args.last[:spec_path] = caller(0)[1]
    register_behaviour(Spec::DSL::BehaviourFactory.create(*args, &block))
  end
