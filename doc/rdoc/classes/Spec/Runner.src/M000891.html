        def run_spec_via_druby(args)
          err, out = StringIO.new, StringIO.new
          out.instance_eval do
            def tty?; true end
          end
          Spec::Runner::DrbCommandLine.run(args, err, out, false, true)
          out.rewind; out.read
        end
