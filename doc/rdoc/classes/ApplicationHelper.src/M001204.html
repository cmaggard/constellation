  def admin_links
    if session[:admin]
      links = "Administer: "
      links += link_to "Technicians", { :controller => '/admin/technician' }
      links += " | "
      links += link_to "Links", { :controller => '/admin/links' }
      links += " | "
      links += link_to "Reports", { :controller => '/report' }
    end # session[:admin]
  end
