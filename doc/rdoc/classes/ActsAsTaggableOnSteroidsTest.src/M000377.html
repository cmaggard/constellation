  def test_case_insensitivity
    assert_difference "Tag.count", 1 do
      Post.create!(:text => "Test", :tag_list => "one")
      Post.create!(:text => "Test", :tag_list => "One")
    end
    
    assert_equal Post.find_tagged_with("Nature"), Post.find_tagged_with("nature")
  end
