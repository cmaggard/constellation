  def test_before_validation
    cust = Customer.new
    cust.first_name = "Don  "
    cust.last_name = "  Giovanni  "
    cust.phone_number = " 803-205-1025"
    assert cust.save
    assert_equal "Don", cust.first_name
    assert_equal "Giovanni", cust.last_name
    assert_equal "803-205-1025", cust.phone_number
  end
