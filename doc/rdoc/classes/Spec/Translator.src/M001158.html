    def translate_dir(from, to)
      FileUtils.mkdir_p(to) unless File.directory?(to)
      Dir["#{from}/*"].each do |sub_from|
        path = sub_from[from.length+1..-1]
        sub_to = File.join(to, path)
        translate(sub_from, sub_to)
      end
    end
