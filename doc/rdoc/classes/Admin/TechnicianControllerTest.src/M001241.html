  def test_create
    num_technicians = Technician.count

    post :create, :technician => {:name => "Dan Embrey", :initials => "DE"}

    assert_response :redirect
    assert_redirected_to :action => 'list'

    assert_equal num_technicians + 1, Technician.count
  end
