      def parse_description(str)
        return str[1..-2] if str =~ /^['"].*['"]$/
        if matches = /^(.*)\s*,\s*['"](.*)['"]$/.match(str)
          return ::Spec::DSL::Description.generate_description(matches[1], matches[2])
        end
        return str
      end
