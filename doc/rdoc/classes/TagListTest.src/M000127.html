  def test_alternative_delimiter
    TagList.delimiter = " "
    
    assert_equal %w(One Two), TagList.from("One Two")
    assert_equal ['One two', 'three', 'four'], TagList.from('"One two" three four')
  ensure
    TagList.delimiter = ","
  end
