  def test_find_customer_js_results
    cody = customers(:cody)
    xhr :post, :find_customer, { :customer => { :first_name => cody.first_name, 
                                                :last_name => "" } }
    
    assert_response :success
    assert_match %r|<a href="/new/find_customer/#{cody.id}">Select</a>|, @response.body
  end
