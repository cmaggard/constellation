    def open(name, attributes = {}, alt_atts = {}, &block)
      text = nil
      if attributes.is_a? String
        text = attributes
        attributes = alt_atts
      end

      puts "<#{name}#{buffer.build_attributes(attributes)}>"
      tab_up
        # Print out either the text (using push_text) or call the block and add an endline
        if text
          puts(text)
        elsif block
          block.call
        end
      tab_down
      puts "</#{name}>"
      nil
    end
