    def self.profile(times, &block)
      # Runs the profiler, collects information
      Profiler__::start_profile
      times.times &block
      Profiler__::stop_profile
      
      # Outputs information to a StringIO, returns result
      io = StringIO.new
      Profiler__::print_profile(io)
      io.pos = 0
      result = io.read
      io.close
      result
    end
