    def push_script(result, tabulation, flattened)
      if flattened
        result = Haml::Helpers.find_and_preserve(result)
      end

      result = result.to_s
      while result[-1] == 10 # \n
        # String#chomp is slow
        result = result[0...-1]
      end
      
      result = result.gsub("\n", "\n#{tabs(tabulation)}")
      push_text result, tabulation
      nil
    end
