      def spec_name_for(io, line_number)
        source  = io.read
        behaviour, behaviour_line = behaviour_at_line(source, line_number)
        example, example_line = example_at_line(source, line_number)
        if behaviour && example && (behaviour_line < example_line)
          "#{behaviour} #{example}"
        elsif behaviour
          behaviour
        else
          nil
        end
      end
